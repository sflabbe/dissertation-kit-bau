\section{Case Study 3: Parameter Optimization}
\label{sec:case3}

\subsection{Problem Description}
\label{subsec:case3_problem}

This case investigates optimal parameters to minimize peak response. The optimization problem is formulated as:
\begin{equation}
\min_{k,c} J(k,c) = \max_t |x(t;k,c)|
\end{equation}
subject to:
\begin{align}
5 \times 10^4 &\leq k \leq 2 \times 10^5 \quad \text{[N/m]} \\
500 &\leq c \leq 2000 \quad \text{[Ns/m]}
\end{align}

where $J$ is the objective function representing peak response, $k$ is stiffness, $c$ is damping, and $x(t)$ is the displacement response.

\subsection{Optimization Algorithm}
\label{subsec:case3_algorithm}

The parameter optimization is performed using a gradient-based approach with finite difference approximations. Algorithm~\ref{alg:optimization} presents the complete procedure.

\begin{algorithm}[htbp]
\caption{Parameter Optimization Algorithm}
\label{alg:optimization}
\begin{algorithmic}[1]
\REQUIRE Initial parameters $k_0$, $c_0$, tolerance $\epsilon = 10^{-4}$, max iterations $N_{\text{max}} = 100$
\ENSURE Optimized parameters $k_{\text{opt}}$, $c_{\text{opt}}$
\STATE Initialize: $k \leftarrow k_0$, $c \leftarrow c_0$, $i \leftarrow 0$
\STATE Compute initial objective: $J_0 \leftarrow \text{EvaluateResponse}(k,c)$
\REPEAT
    \STATE $i \leftarrow i + 1$
    \STATE \textbf{// Compute gradients using finite differences}
    \STATE $\Delta k \leftarrow 0.01 \cdot k$, $\Delta c \leftarrow 0.01 \cdot c$
    \STATE $J_k^+ \leftarrow \text{EvaluateResponse}(k+\Delta k, c)$
    \STATE $J_c^+ \leftarrow \text{EvaluateResponse}(k, c+\Delta c)$
    \STATE $g_k \leftarrow (J_k^+ - J_0)/\Delta k$
    \STATE $g_c \leftarrow (J_c^+ - J_0)/\Delta c$
    \STATE \textbf{// Line search with backtracking}
    \STATE $\alpha \leftarrow 1.0$
    \WHILE{$\alpha > 10^{-6}$}
        \STATE $k_{\text{new}} \leftarrow k - \alpha \cdot g_k$
        \STATE $c_{\text{new}} \leftarrow c - \alpha \cdot g_c$
        \STATE \textbf{// Apply bounds}
        \STATE $k_{\text{new}} \leftarrow \max(5\times10^4, \min(k_{\text{new}}, 2\times10^5))$
        \STATE $c_{\text{new}} \leftarrow \max(500, \min(c_{\text{new}}, 2000))$
        \STATE $J_{\text{new}} \leftarrow \text{EvaluateResponse}(k_{\text{new}}, c_{\text{new}})$
        \IF{$J_{\text{new}} < J_0$}
            \STATE \textbf{break}
        \ENDIF
        \STATE $\alpha \leftarrow 0.5 \cdot \alpha$
    \ENDWHILE
    \STATE \textbf{// Update parameters}
    \STATE $k \leftarrow k_{\text{new}}$, $c \leftarrow c_{\text{new}}$
    \STATE $\Delta J \leftarrow |J_0 - J_{\text{new}}|$
    \STATE $J_0 \leftarrow J_{\text{new}}$
\UNTIL{$\Delta J < \epsilon$ or $i \geq N_{\text{max}}$}
\STATE $k_{\text{opt}} \leftarrow k$, $c_{\text{opt}} \leftarrow c$
\RETURN $k_{\text{opt}}$, $c_{\text{opt}}$
\end{algorithmic}
\end{algorithm}

The algorithm converged after 23 iterations, reducing the peak response by 22.4\%.
